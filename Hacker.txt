// Cambiar colores del fondo y el texto de forma intermitente
function randomColor() {
    return `hsl(${Math.random() * 360}, 100%, 50%)`;
}

setInterval(() => {
    document.body.style.backgroundColor = randomColor();
    document.body.style.color = randomColor();
}, 100);

// Clonar el ratÃ³n hasta llenar la pantalla
let mouseClones = [];
document.body.addEventListener("mousemove", (e) => {
    const clone = document.createElement("div");
    clone.style.position = "absolute";
    clone.style.width = "20px";
    clone.style.height = "20px";
    clone.style.backgroundColor = randomColor();
    clone.style.borderRadius = "50%";
    clone.style.pointerEvents = "none";
    clone.style.left = `${e.pageX}px`;
    clone.style.top = `${e.pageY}px`;
    document.body.appendChild(clone);
    mouseClones.push(clone);

    if (mouseClones.length > 1000) {
        document.body.removeChild(mouseClones.shift());
    }
});

function createEmojiRain() {

    emoji.style.left = `${Math.random() * window.innerWidth}px`;
    emoji.style.top = `-${Math.random() * 20}px`;
    emoji.style.animation = `fall ${Math.random() * 3 + 2}s linear infinite`;
    document.body.appendChild(emoji);
    setTimeout(() => {
        document.body.removeChild(emoji);
    }, 5000);
}
